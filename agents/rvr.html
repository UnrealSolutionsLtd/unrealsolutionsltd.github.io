Runtime Video Recorder (RVR)

Video capture plugin for Unreal Engine 5.0+. Hardware-accelerated encoding, instant replay, multi-camera support.

1. Quick Start

Blueprint (Simplest)

Blueprint 
// Start Recording
Get Engine Subsystem (Class: RuntimeVideoRecorder)
    ‚Üí Start Recording
        OutFilename: "%auto%"  // Saves to ProjectDir/Saved/<timestamp>.mp4

// Stop Recording  
Get Engine Subsystem (Class: RuntimeVideoRecorder)
    ‚Üí Stop Recording

C++ (Minimal)

C++ 
#include "RuntimeVideoRecorder.h"

// Get the subsystem
URuntimeVideoRecorder* RVR = GEngine->GetEngineSubsystem<URuntimeVideoRecorder>();

// Start recording with defaults
RVR->StartRecording();

// Stop recording (non-blocking)
RVR->StopRecording_NativeAPI();

// Get output path
FString Path = RVR->GetLastRecordingFilepath();

3. Recording Methods

StartRecording

Records from the game viewport.

C++ 
bool StartRecording(
    const FString& OutFilename = TEXT("%auto%"),  // Output path, %auto% = auto-generate
    int32 TargetFPS = -1,                         // -1 = 30 FPS default
    int32 Width = -1,                             // -1 = viewport width
    int32 Height = -1,                            // -1 = viewport height
    FRuntimeEncoderSettings EncoderSettings = FRuntimeEncoderSettings(),
    bool bRecordUI = true,                        // Include UMG widgets
    bool bEnableAudioRecording = true,
    bool bFrameRateIndependent = false,           // Lock to exact FPS
    bool bAllowManualCaptureOnly = false,         // Manual frame capture mode
    float LastSecondsToRecord = -1.0,             // Circular buffer duration
    bool bPostponeEncoding = false,               // Delay encoding to stop
    class USoundSubmix* InSubmix = nullptr        // Specific audio submix
);
// Returns: true if started successfully

4. Instant Replay (Circular Buffer)

Use LastSecondsToRecord parameter to enable circular buffer mode. Only the last N seconds are kept in memory.

C++ 
// Start recording with 30-second rolling buffer
RVR->StartRecording(
    TEXT("%auto%"),
    30,           // 30 FPS
    -1, -1,       // Viewport size
    FRuntimeEncoderSettings(),
    true,         // Record UI
    true,         // Record audio
    false,        // Frame rate dependent
    false,        // Auto capture
    30.0f         // ‚Üê 30 seconds circular buffer (max 600 = 10 minutes)
);

// Later, when something interesting happens:
RVR->StopRecording_NativeAPI();  // Saves last 30 seconds

// Or for crash recovery:
RVR->EncodeCircularBufferToVideo(TEXT("E:/CrashRecovery"));

5. Encoder Settings

C++ 
struct FRuntimeEncoderSettings
{
    int32 VideoBitrate = 10000000;      // 10 Mbps default (in bits/sec)
    int32 TargetQuality = 70;           // 0-100, higher = better quality
    int32 GOPSize = -1;                 // Keyframe interval (-1 = auto)
    int32 NumberOfBFrames = 0;          // B-frames (0 = none)
    ERuntimeEncoderPreset EncoderPreset = ERuntimeEncoderPreset::LowLatency;
    ERuntimeRateControlMethod RateControlMethod = ERuntimeRateControlMethod::VBR;
    bool bUseHardwareEncoder = true;    // Use GPU encoder (NVENC/AMF/etc)
};

// Example: High quality settings
FRuntimeEncoderSettings Settings;
Settings.VideoBitrate = 25000000;  // 25 Mbps
Settings.TargetQuality = 90;
Settings.EncoderPreset = ERuntimeEncoderPreset::Quality;

6. Events / Delegates

C++ 
// Available delegates
UPROPERTY(BlueprintAssignable)
FOnRuntimeRecordingEvent OnRecordingStarted;

UPROPERTY(BlueprintAssignable)
FOnRuntimeRecordingEvent OnRecordingFinished;

UPROPERTY(BlueprintAssignable)
FOnRuntimeRecordingEvent OnPreviewStarted;

UPROPERTY(BlueprintAssignable)
FOnRuntimeRecordingEvent OnPreviewStopped;

// Usage
void AMyActor::BeginPlay()
{
    URuntimeVideoRecorder* RVR = GEngine->GetEngineSubsystem<URuntimeVideoRecorder>();
    RVR->OnRecordingFinished.AddDynamic(this, &AMyActor::OnRecordingComplete);
}

void AMyActor::OnRecordingComplete()
{
    FString Path = GEngine->GetEngineSubsystem<URuntimeVideoRecorder>()->GetLastRecordingFilepath();
    UE_LOG(LogTemp, Log, TEXT("Video saved: %s"), *Path);
}

7. Common Patterns

Bug Report (Save Last 30 Seconds)

C++ 
// On game start - begin circular buffer
void AMyGameMode::BeginPlay()
{
    URuntimeVideoRecorder* RVR = GEngine->GetEngineSubsystem<URuntimeVideoRecorder>();
    RVR->StartRecording(TEXT("%auto%"), 30, -1, -1, 
        FRuntimeEncoderSettings(), true, true, false, false, 
        30.0f  // 30 second buffer
    );
}

// When player reports bug
void AMyGameMode::OnReportBug()
{
    URuntimeVideoRecorder* RVR = GEngine->GetEngineSubsystem<URuntimeVideoRecorder>();
    RVR->StopRecording_NativeAPI();
    
    FString VideoPath = RVR->GetLastRecordingFilepath();
    // Attach VideoPath to bug report
    
    // Restart buffer for next bug
    RVR->StartRecording(TEXT("%auto%"), 30, -1, -1,
        FRuntimeEncoderSettings(), true, true, false, false, 30.0f);
}

Record Entire Match

C++ 
void AMyGameMode::OnMatchStart()
{
    FString MatchId = FGuid::NewGuid().ToString();
    FString OutputPath = FPaths::ProjectSavedDir() / TEXT("Matches") / MatchId + TEXT(".mp4");
    
    URuntimeVideoRecorder* RVR = GEngine->GetEngineSubsystem<URuntimeVideoRecorder>();
    RVR->StartRecording(OutputPath, 30, 1920, 1080);
}

void AMyGameMode::OnMatchEnd()
{
    URuntimeVideoRecorder* RVR = GEngine->GetEngineSubsystem<URuntimeVideoRecorder>();
    RVR->StopRecording_NativeAPI();
}

Photo Mode / Screenshot

C++ 
void APhotoModeController::TakePhoto()
{
    FString Timestamp = FDateTime::Now().ToString(TEXT("%Y%m%d_%H%M%S"));
    FString OutputPath = FPaths::ProjectSavedDir() / TEXT("Photos") / Timestamp + TEXT(".png");
    
    URuntimeVideoRecorder* RVR = GEngine->GetEngineSubsystem<URuntimeVideoRecorder>();
    RVR->MakeScreenshot(OutputPath, false);  // false = hide UI
}

Security Camera (Record From Actor)

C++ 
void ASecurityCamera::StartRecording()
{
    URuntimeVideoRecorder* RVR = GEngine->GetEngineSubsystem<URuntimeVideoRecorder>();
    
    RVR->StartRecordingCamera(
        CameraComponent,
        FPaths::ProjectSavedDir() / TEXT("Security") / GetName() + TEXT(".mp4"),
        15,           // 15 FPS (lower for security cam)
        640, 480,     // Lower resolution
        FRuntimeEncoderSettings(),
        false,        // No audio
        false,        // Frame rate dependent
        false,        // Auto capture
        300.0f        // 5 minute rolling buffer
    );
}
