1. Overview

BugIt captures deterministic, frame-aligned repro bundles containing:

  • ‚Ä¢ Video - MP4 recording via RVR circular buffer
  • ‚Ä¢ Inputs - Every keyboard, mouse, gamepad, touch event
  • ‚Ä¢ Logs - Frame-aligned UE_LOG output
  • ‚Ä¢ Timing - Frame context with player/camera positions
  • ‚Ä¢ Hardware - System specs for reproduction
  • ‚Ä¢ Custom Data - Game state, events, markers, screenshots
Dependency: BugIt requires Runtime Video Recorder (RVR) for video capture.

2. Quick Start

C++ Integration

C++ 
// In GameMode or PlayerController
#include "BugItCaptureSubsystem.h"

void AMyGameMode::BeginPlay()
{
    Super::BeginPlay();
    
    // Get the subsystem
    UBugItCaptureSubsystem* BugIt = GetGameInstance()->GetSubsystem<UBugItCaptureSubsystem>();
    
    // Configure capture settings
    FBugItCaptureSettings Settings;
    Settings.BufferDurationSeconds = 60.0f;  // 1 minute buffer
    Settings.TargetFPS = 30;
    Settings.bCaptureInputs = true;
    Settings.bCaptureLogs = true;
    
    // Set build info for tracking
    BugIt->SetBuildInfo(GetBuildInfo());
    
    // Start capture
    BugIt->StartCapture(Settings);
    
    // Bind completion handler
    BugIt->OnBugReportReady.AddDynamic(this, &AMyGameMode::OnBugReportComplete);
}

void AMyGameMode::ReportBug(const FString& Notes)
{
    UBugItCaptureSubsystem* BugIt = GetGameInstance()->GetSubsystem<UBugItCaptureSubsystem>();
    BugIt->ReportBug(Notes, {TEXT("crash"), TEXT("gameplay")});
}

void AMyGameMode::OnBugReportComplete(const FString& BundlePath, const FString& BundleId)
{
    UE_LOG(LogGame, Log, TEXT("Bug report ready: %s"), *BundleId);
    // Upload to BugIt server
    UploadBundle(BundlePath);
}

3. Capture Subsystem API

StartCapture

C++ 
void StartCapture(FBugItCaptureSettings Settings);

struct FBugItCaptureSettings
{
    float BufferDurationSeconds = 30.0f;   // Rolling buffer size (max 600 = 10 min)
    int32 TargetFPS = 30;
    int32 VideoWidth = -1;                  // -1 = viewport
    int32 VideoHeight = -1;
    bool bCaptureInputs = true;             // Capture all input events
    bool bCaptureLogs = true;               // Capture UE_LOG output
    bool bIncludeAudio = true;
    TArray<FName> LogCategories;            // Filter log categories (empty = all)
};

ReportBug

C++ 
void ReportBug(
    const FString& Notes,                   // User description
    const TArray<FString>& Tags             // Categorization tags
);

// Triggers:
// 1. Freeze all rolling buffers (atomic snapshot)
// 2. Encode video from circular buffer
// 3. Write all artifacts to disk
// 4. Calculate content hash
// 5. Fire OnBugReportReady delegate

C++ 
void SetBuildInfo(FBugItBuildInfo Info);

4. Custom Game Data

Inject game-specific context to complement system-level capture:

C++ 
UBugItCaptureSubsystem* BugIt = GetGameInstance()->GetSubsystem<UBugItCaptureSubsystem>();

// Persistent properties (included in every state snapshot)
BugIt->SetPersistentProperty(TEXT("PlayerClass"), TEXT("Warrior"));
BugIt->SetPersistentPropertyInt(TEXT("Difficulty"), 3);

// Log custom events at important moments
BugIt->LogCustomEvent(
    TEXT("BossSpawned"),           // Event name
    TEXT("Combat"),                // Category
    1,                             // Severity (0=Info, 1=Warning, 2=Error)
    {                              // Custom data map
        {TEXT("BossId"), TEXT("dragon_01")},
        {TEXT("PlayerHealth"), TEXT("85")}
    }
);

// Push game state each frame (or on significant changes)
TMap<FString, FString> State;
State.Add(TEXT("Health"), FString::FromInt(Player->Health));
State.Add(TEXT("Ammo"), FString::FromInt(Player->Ammo));
State.Add(TEXT("QuestId"), CurrentQuest->GetId());
BugIt->PushGameState(State);

// Add timeline markers at key moments
BugIt->AddMarker(TEXT("Boss Phase 2"), TEXT("Dragon entered rage mode"), FColor::Red);

// Take annotated screenshot
BugIt->TakeAnnotatedScreenshot(
    TEXT("UI glitch visible here"),    // Caption
    {TEXT("ui"), TEXT("glitch")},      // Tags
    {{TEXT("MenuName"), TEXT("Inventory")}}  // Metadata
);

7. Capture Triggers

BugIt supports multiple trigger types for automatic bug capture:

Trigger Description Example
Manual User presses button ReportBug(Notes, Tags)
Gameplay Event On death, level fail, etc. Call ReportBug in OnPlayerDeath
Timer Periodic auto-capture Every 5 minutes during playtest
Performance FPS drops below threshold Auto-capture when FPS < 20
Error/Assert On ensure/check failure Bind to FCoreDelegates

Performance Trigger Example

C++ 
void AMyGameMode::Tick(float DeltaTime)
{
    Super::Tick(DeltaTime);
    
    float CurrentFPS = 1.0f / DeltaTime;
    
    if (CurrentFPS < 20.0f && !bLowFPSReported)
    {
        bLowFPSReported = true;
        
        UBugItCaptureSubsystem* BugIt = GetGameInstance()->GetSubsystem<UBugItCaptureSubsystem>();
        BugIt->LogCustomEvent(TEXT("LowFPS"), TEXT("Performance"), 1, 
            {{TEXT("FPS"), FString::Printf(TEXT("%.1f"), CurrentFPS)}});
        BugIt->ReportBug(
            FString::Printf(TEXT("Auto-captured: FPS dropped to %.1f"), CurrentFPS),
            {TEXT("performance"), TEXT("auto")}
        );
    }
}
